# useRef로 컴포넌트 안의 변수 만들기
`useRef` Hook의 기능: DOM을 선택하는 용도, **컴포넌트 안에서 변수를 관리**  
`useRef`로 관리하는 변수는 값이 바뀌어도 컴포넌트가 리렌더링 되지 않는다.  
=> 변수 설정 후 바로 조회할 수 있다.

`useRef`를 사용하여 배열에서 새로운 항목을 추가할 때, 고유 id를 관리하는 용도로 변수를 관리한다.

**UserList.js**
```JavaScript
import React from 'react';

function User({ user }) {
  return (
    <div>
      <b>{user.username}</b> <span>({user.email})</span>
    </div>
  );
}

function UserList({ users }) {
  return (
    <div>
      {users.map(user => (
        <User user={user} key={user.id} />
      ))}
    </div>
  );
}

export default UserList;
```


**App.js**
```JavaScript
import React, { useRef } from 'react';
import UserList from './UserList';

function App() {
  const users = [
    {
      id: 1,
      username: 'velopert',
      email: 'public.velopert@gmail.com'
    },
    {
      id: 2,
      username: 'tester',
      email: 'tester@example.com'
    },
    {
      id: 3,
      username: 'liz',
      email: 'liz@example.com'
    }
  ];

  const nextId = useRef(4); //useRef()를 사용할 때 파라미터(4)를 넣어주면 .current의 기본값
  const onCreate = () => { //배열에 새 항목을 추가할 때 id가 4인 항목 추가
    
    nextId.current += 1;
  };

  return <UserList users={users} />;
}

export default App;

```
