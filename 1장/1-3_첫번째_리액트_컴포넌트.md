```jsx
//Hello.js

import React from 'react';

function Hello(){
return (
<div> 안녕하세요</div>
); //return으로 내보내주어야 한다.
}
export defalut Hello; // Hello 라는 컴포넌트 내보냄, 다른컴포넌트에서 불러올 수 있다.

```

```jsx

// 구현할 컴포넌트 파일.
//app.js에 구현
import React from 'react';
import Hello from './Hello';

function App(){
	return (
		<div>
			<Hello/> //div Hello 컴포넌트 불러오려면 <Hello/>,
		</div>
	};
}

export default App;  // App에 컴포넌트들 배열하고, 이 App을 index에서 받는다

//index.js
//
import React from 'react'; //리액트 컴포넌트 만들기
import ReactDOM from 'react-dom'; //dom 불러오기
import './index.css'; //css파일
import App from './App'; //컴포넌트 배열한 App 불러오기
import * as serviceWorker from './serviceWorker'; //뭔지 모르겠다

ReactDOM.render(<App />, document.getElementById('root'));// id가 root인 부분에 컴포넌트를 렌더링 할것이다.

// If you want your app to work offline and load faster, you can change
// unregister() to register() below. Note this comes with some pitfalls.
// Learn more about service workers: https://bit.ly/CRA-PWA
serviceWorker.unregister();.

```
